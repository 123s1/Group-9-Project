# 支付财务模块 ER 图

```mermaid
erDiagram
    payment_record {
        BIGINT id PK
        BIGINT order_id FK UK "关联订单"
        BIGINT parent_id FK "支付方家长"
        BIGINT driver_id "收款接送员"
        DECIMAL amount "支付金额"
        DECIMAL discount_amount "优惠金额"
        DECIMAL actual_amount "实付金额"
        BIGINT coupon_id "使用的优惠券"
        TINYINT payment_method "1-微信 2-支付宝 3-钱包余额"
        VARCHAR transaction_id "第三方支付流水号"
        TINYINT trade_type "1-即时支付 2-担保交易"
        TINYINT status "0-待支付 1-支付中 2-已支付 3-已退款"
        DATETIME paid_at "支付时间"
        DATETIME created_at "创建时间"
        DATETIME updated_at "更新时间"
    }

    refund_record {
        BIGINT id PK
        BIGINT order_id FK "关联订单"
        BIGINT payment_id FK "原支付记录"
        BIGINT parent_id FK "退款接收方"
        DECIMAL refund_amount "退款金额"
        TINYINT refund_type "1-用户取消 2-系统取消 3-客服退款"
        VARCHAR reason "退款原因"
        VARCHAR reject_reason "驳回原因"
        TINYINT status "0-待审核 1-通过 2-驳回 3-退款中 4-已退款 5-失败"
        BIGINT handler_id "处理人"
        DATETIME handled_at "处理时间"
        DATETIME refund_at "退款到账时间"
        DATETIME created_at "创建时间"
        DATETIME updated_at "更新时间"
    }

    invoice {
        BIGINT id PK
        BIGINT order_id FK "关联订单"
        BIGINT parent_id FK "申请家长"
        TINYINT invoice_type "1-普通电子 2-增值税专用 3-增值税普通"
        TINYINT title_type "1-个人 2-企业"
        VARCHAR title_name "发票抬头"
        VARCHAR tax_number "税号"
        VARCHAR registered_address "注册地址"
        VARCHAR registered_phone "注册电话"
        VARCHAR bank_name "开户银行"
        VARCHAR bank_account "银行账号"
        DECIMAL billing_amount "开票金额"
        TINYINT status "0-待审核 1-已开票 2-已作废 3-已驳回"
        VARCHAR invoice_url "发票文件 URL"
        VARCHAR invoice_number "发票编号"
        BIGINT handler_id "处理人"
        DATETIME handled_at "处理时间"
        DATETIME created_at "创建时间"
        DATETIME updated_at "更新时间"
    }

    driver_settlement {
        BIGINT id PK
        BIGINT driver_id FK "接送员"
        VARCHAR settlement_period "结算周期"
        INT order_count "本期完成订单数"
        DECIMAL gross_amount "本期总收入"
        DECIMAL commission_rate "平台抽成比例"
        DECIMAL commission_amount "平台抽成金额"
        DECIMAL net_amount "净收入"
        DECIMAL withdrawable_amount "可提现金额"
        DECIMAL withdrawn_amount "已提现金额"
        DECIMAL penalty_amount "扣款/罚款"
        DECIMAL bonus_amount "奖励金额"
        TINYINT status "0-待对账 1-已确认 2-结算完成 3-已冻结"
        DATETIME confirm_at "确认时间"
        DATETIME settled_at "结算完成时间"
        VARCHAR remark "备注"
        DATETIME created_at "创建时间"
        DATETIME updated_at "更新时间"
    }

    withdrawal_request {
        BIGINT id PK
        BIGINT driver_id FK "申请接送员"
        DECIMAL amount "提现金额"
        TINYINT withdraw_method "1-微信 2-支付宝 3-银行卡"
        VARCHAR account_name "账户姓名"
        VARCHAR account_number "账号"
        VARCHAR bank_name "开户行"
        VARCHAR bank_branch "支行名称"
        TINYINT status "0-待审核 1-通过 2-驳回 3-打款中 4-已到账 5-失败"
        VARCHAR reject_reason "驳回原因"
        BIGINT handler_id "处理人"
        DATETIME handled_at "处理时间"
        VARCHAR transaction_id "第三方打款流水号"
        DATETIME arrived_at "到账时间"
        DATETIME created_at "创建时间"
        DATETIME updated_at "更新时间"
    }

    orders ||--o| payment_record : "支付记录"
    payment_record ||--o{ refund_record : "退款记录"
    orders ||--o{ refund_record : "退款订单"
    orders ||--o{ invoice : "发票"
    sys_user ||--o{ driver_settlement : "接送员结算"
    sys_user ||--o{ withdrawal_request : "提现申请"
```
