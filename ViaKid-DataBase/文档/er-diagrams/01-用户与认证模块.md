# 01-用户与认证模块 ER 图

```mermaid
erDiagram
    drivers {
        uuid id PK "UUID主键"
        nvarchar phone UK "独立手机号"
        nvarchar password_hash "bcrypt密码"
        nvarchar name "姓名"
        nvarchar gender "性别"
        date birthday "出生日期"
        nvarchar avatar_url "头像"
        nvarchar emergency_contact "紧急联系人"
        nvarchar emergency_phone "紧急联系电话"
        tinyint verification_level "0-未认证 1-基础 2-身份 3-驾照 4-车辆 5-完全"
        decimal avg_rating "平均评分"
        int total_orders "完成订单总数"
        nvarchar status "pending/approved/rejected/probation/formal/suspended"
        tinyint online_status "0-离线 1-空闲 2-接单中 3-服务中"
        datetime2 last_online_at "最后上线时间"
        bit is_deleted "软删除"
        datetime2 created_at
        datetime2 updated_at
    }

    parents {
        uuid id PK "UUID主键（阶段一模拟）"
        nvarchar phone UK "独立手机号"
        nvarchar password_hash "bcrypt密码"
        nvarchar name "姓名"
        nvarchar avatar_url "头像"
        nvarchar gender "性别"
        date birthday "出生日期"
        nvarchar status "状态"
        bit is_deleted "软删除"
        datetime2 created_at
        datetime2 updated_at
    }

    admins {
        uuid id PK "UUID主键"
        nvarchar phone UK "独立手机号"
        nvarchar password_hash "bcrypt密码"
        nvarchar username UK "管理员登录名"
        nvarchar nickname "昵称"
        nvarchar email "邮箱"
        nvarchar department "部门"
        nvarchar role "super_admin/operator/finance"
        nvarchar status "状态"
        datetime2 created_at
        datetime2 updated_at
    }

    qualifications {
        uuid id PK "UUID主键"
        uuid driver_id FK UK "关联drivers，driver_id+qual_type唯一"
        tinyint qual_type "1-身份证 2-驾驶证 3-行驶证 4-无犯罪记录 5-健康证明"
        nvarchar qual_name "资质名称"
        tinyint status "0-待提交 1-审核中 2-通过 3-驳回"
        nvarchar reject_reason "驳回原因"
        date expire_date "到期日"
        datetime2 verify_time "审核通过时间"
        uuid verify_admin_id "审核管理员"
        datetime2 created_at
        datetime2 updated_at
    }

    qualification_documents {
        uuid id PK "UUID主键"
        uuid qualification_id FK "关联qualifications"
        nvarchar doc_name "材料名称"
        nvarchar doc_url "材料URL"
        nvarchar doc_type "IMAGE/PDF/VIDEO"
        int file_size "文件大小"
        int sort "排序"
        datetime2 created_at
    }

    schedules {
        uuid driver_id PK, FK "关联drivers"
        nvarchar time_slots "JSON排班数据"
        nvarchar work_days "JSON工作日"
        nvarchar unavailable_dates "JSON不可用日期"
        int max_orders_per_day "日最大接单数"
        datetime2 updated_at
    }

    sms_codes {
        uuid id PK "UUID主键"
        nvarchar phone "手机号"
        nvarchar code "验证码"
        nvarchar type "login/register/reset_password"
        datetime2 expired_at "过期时间"
        bit used "是否已用"
        datetime2 created_at
    }

    refresh_tokens {
        uuid id PK "UUID主键"
        uuid driver_id FK "关联drivers"
        nvarchar token "JWT刷新令牌"
        datetime2 expired_at "过期时间"
        datetime2 created_at
    }

    drivers ||--o{ qualifications : "1:N"
    drivers ||--o| schedules : "1:1"
    drivers ||--o{ refresh_tokens : "1:N"
    qualifications ||--o{ qualification_documents : "1:N"
```
