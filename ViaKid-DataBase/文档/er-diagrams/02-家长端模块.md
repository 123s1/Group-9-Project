# 家长端模块 ER 图

```mermaid
erDiagram
    parent_profile {
        BIGINT id PK
        BIGINT user_id FK UK "关联 sys_user"
        VARCHAR real_name "真实姓名"
        VARCHAR id_card "身份证号"
        VARCHAR avatar_url "头像 URL"
        TINYINT gender "0-未知 1-男 2-女"
        DATE birthday "生日"
        VARCHAR emergency_name "紧急联系人姓名"
        VARCHAR emergency_phone "紧急联系人电话"
        VARCHAR emergency_relation "与紧急联系人的关系"
        BIGINT member_level_id FK "会员等级"
        INT total_points "累计积分"
        INT available_points "可用积分"
        DATETIME created_at "创建时间"
        DATETIME updated_at "更新时间"
    }

    child {
        BIGINT id PK
        BIGINT parent_id FK "关联家长"
        VARCHAR child_name "孩子姓名"
        VARCHAR id_card "身份证号"
        TINYINT gender "0-未知 1-男 2-女"
        DATE birthday "出生日期"
        VARCHAR avatar_url "头像 URL"
        VARCHAR school_name "学校名称"
        VARCHAR grade "年级"
        VARCHAR class_name "班级"
        VARCHAR pickup_code UK "接送码"
        TEXT health_notes "健康备注"
        DECIMAL height_cm "身高（cm）"
        DECIMAL weight_kg "体重（kg）"
        VARCHAR photo_url "孩子照片 URL"
        TINYINT is_deleted "0-未删除 1-已删除"
        DATETIME created_at "创建时间"
        DATETIME updated_at "更新时间"
        DATETIME deleted_at "逻辑删除时间"
    }

    emergency_contact {
        BIGINT id PK
        BIGINT parent_id FK "关联家长"
        VARCHAR contact_name "联系人姓名"
        VARCHAR contact_phone "联系人电话"
        VARCHAR contact_relation "与孩子关系"
        TINYINT is_default "是否为默认联系人"
        DATETIME created_at "创建时间"
    }

    address_book {
        BIGINT id PK
        BIGINT parent_id FK "关联家长"
        VARCHAR address_name "地址名称"
        TINYINT address_type "1-家 2-学校 3-公司 4-其他"
        VARCHAR full_address "详细地址"
        DECIMAL latitude "纬度"
        DECIMAL longitude "经度"
        VARCHAR contact_name "联系人姓名"
        VARCHAR contact_phone "联系人电话"
        TINYINT is_default "是否为默认地址"
        DATETIME created_at "创建时间"
        DATETIME updated_at "更新时间"
    }

    wallet {
        BIGINT id PK
        BIGINT user_id FK UK "关联 sys_user"
        DECIMAL balance "余额"
        DECIMAL frozen_amount "冻结金额"
        DECIMAL total_recharge "累计充值"
        DECIMAL total_consume "累计消费"
        VARCHAR password "支付密码"
        DATETIME updated_at "更新时间"
    }

    wallet_transaction {
        BIGINT id PK
        BIGINT wallet_id FK "关联钱包"
        TINYINT type "1-充值 2-消费 3-退款 4-提现 5-奖励"
        DECIMAL amount "变动金额"
        DECIMAL balance_before "变动前余额"
        DECIMAL balance_after "变动后余额"
        BIGINT related_order_id "关联订单 ID"
        BIGINT related_payment_id "关联支付记录 ID"
        VARCHAR description "流水描述"
        JSON extend_json "扩展数据"
        DATETIME created_at "创建时间"
    }

    member_level {
        BIGINT id PK
        VARCHAR level_code UK "等级代码"
        VARCHAR level_name "等级名称"
        VARCHAR level_icon "等级图标 URL"
        INT min_points "最低积分门槛"
        INT max_points "最高积分门槛"
        DECIMAL discount_rate "折扣率"
        DECIMAL points_rate "积分倍率"
        TEXT benefits "权益描述 JSON"
        INT sort "排序"
        TINYINT status "1-启用 2-禁用"
        DATETIME created_at "创建时间"
    }

    coupon {
        BIGINT id PK
        BIGINT user_id FK "归属用户"
        VARCHAR coupon_code UK "优惠券码"
        VARCHAR name "优惠券名称"
        TINYINT type "1-满减券 2-折扣券 3-立减券"
        DECIMAL use_threshold "使用门槛"
        DECIMAL discount_value "优惠值"
        DECIMAL max_discount_amount "最高优惠金额"
        DATETIME valid_from "生效时间"
        DATETIME valid_until "过期时间"
        TINYINT status "0-未使用 1-已使用 2-已过期 3-已禁用"
        DATETIME used_at "使用时间"
        BIGINT order_id "关联使用订单"
        VARCHAR source "来源"
        DATETIME created_at "创建时间"
    }

    parent_profile ||--o{ child : "拥有孩子"
    parent_profile ||--o{ emergency_contact : "拥有紧急联系人"
    parent_profile ||--o{ address_book : "拥有地址"
    parent_profile }o--o| member_level : "会员等级"
    wallet ||--o{ wallet_transaction : "流水记录"
    coupon }o--|| sys_user : "归属用户"
```
