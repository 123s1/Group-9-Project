# 03-接送员端模块 ER 图

```mermaid
erDiagram
    drivers ||--o| qualifications : "1:1"
    qualifications ||--o{ qualification_documents : "1:N"
    drivers ||--o| schedules : "1:1"
    drivers ||--o{ course_progress : "1:N"
    course_progress }o--|| courses : "N:1"
    courses ||--o{ exam_questions : "1:N"
    exam_questions ||--o{ exam_options : "1:N"
    drivers ||--o{ exam_results : "1:N"
    exam_results }o--|| courses : "N:1"
    drivers ||--o{ fixed_route : "1:N"

    qualifications {
        uuid id PK
        uuid driver_id FK UK
        tinyint qual_type "1-5"
        nvarchar qual_name
        tinyint status "0-待提交 1-审核中 2-通过 3-驳回"
        nvarchar reject_reason
        date expire_date
        datetime2 verify_time
        uuid verify_admin_id
        datetime2 created_at
        datetime2 updated_at
    }

    qualification_documents {
        uuid id PK
        uuid qualification_id FK
        nvarchar doc_name
        nvarchar doc_url
        nvarchar doc_type "IMAGE/PDF/VIDEO"
        int file_size
        int sort
        datetime2 created_at
    }

    schedules {
        uuid driver_id PK, FK
        nvarchar time_slots "JSON"
        nvarchar work_days "JSON"
        nvarchar unavailable_dates "JSON"
        int max_orders_per_day
        datetime2 updated_at
    }

    courses {
        nvarchar id PK "C001"
        nvarchar title "课程名称"
        nvarchar description "课程描述"
        nvarchar cover_url "封面图"
        nvarchar content_type "video/article/livestream"
        nvarchar content_url "内容URL"
        nvarchar duration "时长描述"
        nvarchar course_type "safety/service/emergency"
        bit is_required "是否必修"
        tinyint pass_score "及格分"
        int sort_order "排序"
        nvarchar status "active/disabled"
        datetime2 created_at
        datetime2 updated_at
    }

    course_progress {
        uuid driver_id PK, FK
        nvarchar course_id PK, FK
        nvarchar status "not_started/in_progress/completed/failed"
        tinyint progress "0-100"
        bigint last_position "视频位置(秒)"
        datetime2 enrolled_at "报名时间"
        datetime2 completed_at "完成时间"
        datetime2 updated_at
    }

    exam_questions {
        nvarchar id PK "Q001"
        nvarchar course_id "关联课程（可空）"
        nvarchar type "single_choice/multiple_choice/true_false"
        nvarchar content "题目内容"
        nvarchar explanation "答案解析"
        int sort_order "排序"
        datetime2 created_at
    }

    exam_options {
        nvarchar id PK "O001"
        nvarchar question_id FK
        nvarchar option_key "A/B/C/D"
        nvarchar content "选项内容"
        bit is_correct "是否正确"
        int sort "排序"
    }

    exam_results {
        uuid id PK
        uuid driver_id FK
        nvarchar course_id "关联课程"
        int total_questions "总题数"
        int correct_count "答对数"
        int score "考试成绩"
        bit passed "是否通过"
        int exam_duration "答题用时(秒)"
        nvarchar answers_json "答题详情"
        datetime2 submitted_at "提交时间"
    }

    fixed_route {
        uuid id PK
        uuid driver_id FK
        nvarchar route_name "线路名称"
        tinyint route_type "1-上学 2-放学 3-往返"
        nvarchar start_point "起点地址"
        decimal start_lat "起点纬度"
        decimal start_lng "起点经度"
        nvarchar end_point "终点地址"
        decimal end_lat "终点纬度"
        decimal end_lng "终点经度"
        nvarchar route_points "JSON途经点"
        decimal total_distance "总距离(km)"
        int estimated_duration "预计时长(分钟)"
        nvarchar week_days "运营日 1,2,3,4,5"
        nvarchar departure_time "发车时间"
        tinyint status "1-启用 2-停用"
        datetime2 created_at
        datetime2 updated_at
    }
```
